Component:
Template + Class + Decorator

Decorators:
https://v2.angular.io/docs/ts/latest/api/core/index/Component-decorator.html

import {Component} from "@angular/core"

@Component({ // decorator
  selector: 'my-app',
  template: `<h1>Hello {{name}}</h1>` // template
})
export class AppComponent { // class
  name: string = "Angular";
}
